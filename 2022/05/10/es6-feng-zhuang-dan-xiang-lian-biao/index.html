ES6封装单向链表

简介链表与数组的区别

数组：

  • 在内存空间中，数组是连续的一块区域
  • 数组声明时要声明大小，可能造成空间的浪费，也可能会因为空间不足而作扩容，效率下降
  • 可以通过下标值来访问数据，随机读取效率较高。
  • 插入数据和删除数据效率低，从中间或前面进行插入或删除操作时，后面的元素要依次位移，效率低下

链表

  • 在内存中，链表中的元素可以不是连续的
  • 每个数据中都存储了下一个数据的位置，可以通过上一个元素查询下一个元素
  • 增加删除数据变得简单
  • 没有下标值，查询必须从头开始查，查询效率低下
  • 无需指定大小，随意更改长度

两者不同操作的时间复杂度

操作 数组 链表
读取 O(1) O(n)
插入 O(n) O(1)
删除 O(n) O(1)

ES6实现单链表

介绍

一般单向链表需要实现以下基本方法

  • append(value) 在最后追加元素
  • insert(pos, value) 在指定位置插入元素
  • get(pos) 获取指定位置元素的值
  • indexOf(value) 获取指定值的元素的位置
  • update(pos, value) 更新指定元素的值
  • removeAt(pos) 删除指定位置的元素
  • remove(value) 删除指定值的元素
  • isEmpty() 判断链表是否为空
  • size() 返回链表的长度

初始化Node和链表

class Node {
      constructor(item) {
        this.item = item
        this.next = null
      }
    }
class LinkedList {
	  constructor() {
        this.head = null
        this.last = null
        this.length = 0
      }
    }

append()方法实现

append追加元素有两种情况，1，链表为空。2，链表不为空

append(value) {
        const node = new Node(value)
        if (!this.head) {
          this.head = node
          this.last = node
        } else {
          this.last.next = node
          this.last = node
        }
        this.length++
      }

insert()方法实现

insert插入元素时存在三种情况，1，向链表最前面插入。2，向链表尾部插入，直接调用append()方法即可。3，向链表中间插入

insert(pos, value) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length) {
          return false
        }
        const node = new Node(value)
        if (pos === 0) {
          if (this.head) {
            this.append(value)
          } else {
            node.next = this.head
            this.head = node
            this.length++
          }
        } else if (pos === this.length) {
          this.append(value)
        } else {
          let prev = null
          let current = this.head
          let i = 0
          while (i < pos) {
            prev = current
            current = current.next
            i++
          }
          prev.next = node
          node.next = current
          this.length++
        }
        return true
      }

get方法实现

get(pos) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length - 1) {
          return false
        }
        let current = this.head
        let i = 0
        while (i < pos) {
          current = current.next
          i++
        }
        return current.item
      }

indexOf方法实现

indexOf(value) {
        let current = this.head
        let i = 0
        while (i < this.length) {
          if (current.item === value) {
            return i
          }
          current = current.next
          i++
        }
        return -1
      }

removeAt方法实现

removeAt方法与insert方法一样，存在三种情况：1，从开始移除。2，从最后移除。3，从中间移除

removeAt(pos) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length - 1) {
          return false
        }
        if (pos === 0) {
          this.head = this.head.next
        } else {
          let prev = null
          let current = this.head
          let i = 0
          while (i < pos) {
            prev = current
            current = current.next
            i++
          }
          prev.next = current.next
          if (pos === this.length - 1) {
            this.last = prev
          }
        }
        this.length--
        return true
      }

update方法实现

update方法直接调用 removeAt方法和insert方法即可完成

update(pos, value) {
        let res_remove = this.removeAt(pos)
        let res_insert = this.insert(pos, value)
        return res_remove && res_insert
      }

remove方法实现

remove方法直接调用 indexOf方法和removeAt方法即可完成

remove(value) {
        let index=this.indexOf(value)
        return this.removeAt(index)
      }

isEmpty和size方法实现

这两个方法实现比较简单，代码如下

isEmpty() {
        return this.length === 0
      }
size() {
        return this.length
      }

class Node {
      constructor(item) {
        this.item = item
        this.next = null
      }
    }
    class LinkedList {
      constructor() {
        this.head = null
        this.last = null
        this.length = 0
      }
      append(value) {
        const node = new Node(value)
        if (!this.head) {
          this.head = node
          this.last = node
        } else {
          this.last.next = node
          this.last = node
        }
        this.length++
      }
      insert(pos, value) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length) {
          return false
        }
        const node = new Node(value)
        if (pos === 0) {
          node.next = this.head
          this.head = node
          this.length++
        } else if (pos === this.length) {
          this.append(value)
        } else {
          let prev = null
          let current = this.head
          let i = 0
          while (i < pos) {
            prev = current
            current = current.next
            i++
          }
          prev.next = node
          node.next = current
          this.length++
        }
        return true
      }
      get(pos) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length - 1) {
          return false
        }
        let current = this.head
        let i = 0
        while (i < pos) {
          current = current.next
          i++
        }
        return current.item
      }
      indexOf(value) {
        let current = this.head
        let i = 0
        while (i < this.length) {
          if (current.item === value) {
            return i
          }
          current = current.next
          i++
        }
        return -1
      }
      update(pos, value) {
        let res_remove = this.removeAt(pos)
        let res_insert = this.insert(pos, value)
        return res_remove && res_insert
      }
      removeAt(pos) {
        // 判断位置溢出
        if (pos < 0 || pos > this.length - 1) {
          return false
        }
        if (pos === 0) {
          this.head = this.head.next
        } else {
          let prev = null
          let current = this.head
          let i = 0
          while (i < pos) {
            prev = current
            current = current.next
            i++
          }
          prev.next = current.next
          if (pos === this.length - 1) {
            this.last = prev
          }
        }
        this.length--
        return true
      }
      remove(value) {
        let index=this.indexOf(value)
        return this.removeAt(index)
      }
      isEmpty() {
        return this.length === 0
      }
      size() {
        return this.length
      }
    }
