ES6排序算法之冒泡,选择,插入,希尔,快排,归并

冒泡排序

冒泡排序，对数组的每一对相邻元素，对比这两个元素，如果前面的元素大于后面的元素，则交换这两个元素，然后对之后的相邻元素重复以上操作，到最后，最大的元素将出现在数组末尾

对未排序的部分重复n次以上操作，就完成了冒泡排序

算法实现：

function bubbleSort(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = 0; j < arr.length - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
      }
    }
  }
}

测试代码：

let arr = [23, 45, 22, 9, 81, 62]
bubbleSort(arr)
alert(arr.join('-'))

选择排序

依次找到未排序的部分的最小值，把它放在最开始的位置。直到所有元素均以排序

算法实现：

function selectionSort(arr) {
  let min = 0
  for (let i = 0; i < arr.length; i++) {
    min = i
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[min]) {
        min = j
      }
    }
    console.log(min)
    let temp = arr[min]
    arr[min] = arr[i]
    arr[i] = temp
  }
}

插入排序

将第一待排序序列第一个元素看做一个有序序列，把第二个元素到最后一个元素当成是未排序序列。

从头到尾依次扫描未排序序列，将扫描到的每个元素插入有序序列的适当位置。

算法实现：

function insertionSort(arr) {
      for (let i = 1; i < arr.length; i++) {
        let curr = arr[i]
        let j = i
        while (j > 0 && curr < arr[j - 1]) {
          arr[j] = arr[j - 1]
          j--
        }
        arr[j] = curr
      }
    }

测试代码：

let arr = [23, 45, 22, 9, 81, 62]
// bubbleSort(arr)
// selectionSort(arr)
insertionSort(arr)
alert(arr.join('-'))

希尔排序

希尔排序是对插入排序的优化，采用分而治之的思想提高了插入排序的效率。

先将整个待排序的记录序列分割成为若干子序列分别进行直接插入排序，待整个序列中的记录”基本有序”时，再对全体记录进行依次直接插入排序。希尔排序的效率与选择的增量序列有关，我们直接选择最简单的除2序列。

算法实现：

function shellSort(arr) {
  let gap = Math.floor(arr.length / 2)
  while (gap >= 1) {
    for (let i = gap; i < arr.length; i++) {
      let curr = arr[i]
      let j = i
      while (j >= gap && curr < arr[j - gap]) {
        arr[j] = arr[j - gap]
        j -= gap
      }
      arr[j] = curr
    }
    gap = Math.floor(gap / 2)
  }
}

测试代码：

let arr = [23, 45, 22, 9, 81, 62]
// bubbleSort(arr)
// selectionSort(arr)
// insertionSort(arr)
shellSort(arr)
alert(arr.join('-'))

快速排序

  1. 从数列中挑出一个元素，称为 “基准”（pivot）。
  2. 重新排序数列，所有元素比基准值小的摆放在基准前面，所有元素比基准值大的摆在基准的后面。
  3. 递归地把小于基准值元素的子数列和大于基准值元素的子数列排序；

这里以第一个元素为基准进行算法实现：

function quickSort(arr) {
  quick(arr, 0, arr.length - 1)
}

function quick(arr, left, right) {
  if (left >= right) {
    return
  }
  let pivot = arr[left]
  let start=left
  let end=right
  while (left < right) {
    while (arr[right] > pivot && left < right) {
      right--
    }
    arr[left] = arr[right]
    while (arr[left] < pivot && left < right) {
      left++
    }
    arr[right]=arr[left]
  }
  arr[left]=pivot
  quick(arr,start,left-1)
  quick(arr,left+1,end)
}

测试代码：

let arr = [23, 45, 22, 9, 81, 62]
// bubbleSort(arr)
// selectionSort(arr)
// insertionSort(arr)
// shellSort(arr)
quickSort(arr)
alert(arr.join('-'))

归并排序

归并排序和核心思想也是分而治之，对于合并长度总和为n的m个有序数组，算法复杂度为O(n)，归并排序就是做了log(n)次的合并长度总和为n的m个有序数组。

算法实现：

function mergeSort(arr){
  if(arr.length<2){
    return arr
  }
  let middle=Math.floor(arr.length/2)
  let left=arr.slice(0,middle)
  let right=arr.slice(middle)
  return merge(mergeSort(left),mergeSort(right))
}

function merge(left,right){
  let res=[]
  while(left.length!==0&&right.length!==0){
    if(left[0]<right[0]){
      res.push(left.shift())
    }else{
      res.push(right.shift())
    }
  }
  return [...res,...left,...right]
}

测试代码：

let arr = [23, 45, 22, 9, 81, 62]
// bubbleSort(arr)
// selectionSort(arr)
// insertionSort(arr)
// shellSort(arr)
// quickSort(arr)
// alert(arr.join('-'))
alert(mergeSort(arr).join('-'))
